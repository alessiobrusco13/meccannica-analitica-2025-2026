\chapter{Equazioni di Hamilton}

Le equazioni trattate in questo capitolo vengono dette \emph{equazioni di Hamilton} o \emph{equazioni canoniche della meccanica} e il loro sviluppo è datato al 1834.

\section{Trasformata di Legendre}

Consideriamo il calcolo del differenziale di una funzione di due variabili $f(x,y)$:
\begin{equation*}
    \dd{f} = \pdv{f}{x} \dd{x} + \pdv{f}{y} \dd{y} = u \dd{x} + v \dd{y}.
\end{equation*}
Talvolta può risultare utile scrivere il differenziale di $f$ in termini di $u$ e $y$ invece che di $x$ e $y$. Attraverso la regola di Leibniz si ha
\begin{equation*}
    \dd{f} = \dd{(ux)} - x \dd{u} + \dd{(vy)} - y \dd{v} = \dd{(ux)} - x \dd{u} + v \dd{y},
\end{equation*}
e portando al primo membro si ha
\begin{equation*}
    \dd{(f - ux)} = - x \dd{u} + v \dd{y}.
\end{equation*}
Ponendo $g(u,y) \defeq f(x,y) - ux$ si ha che
\begin{equation*}
    \dd{g} = - x \dd{u} + v \dd{y},
\end{equation*}
avendo invertito la dipendenza di $f$ da $x$ a $u$, ottenendo la \emph{trasformata di Legendre} di $f$.

\begin{definition}[Trasformata di Legendre]
Sia $f(x,y)$ una funzione di due variabili. La \emph{trasformata di Legendre} di $f$ è la funzione $g(u,y) = f(x,y) - ux$ ottenuta invertendo la dipendenza di $f$ da $x$ a $u$.
    \begin{equation*}
        u = \pdv{f}{x} \iff x = - \pdv{g}{u}.
    \end{equation*}
\end{definition}


\section{Derivazione delle equazioni di Hamilton}
Partiamo dalle equazioni di Lagrange
\begin{equation*}
    \dv{t} \pdv{L}{\dot{q}_i} - \pdv{L}{q_i} = 0.
\end{equation*}
Le equazioni possono essere riscritte in termini dei momenti coniugati
\begin{equation}
    \label{eq:momenticoniugati}
    p_i = \pdv{L}{\dot{q}_i}, \qquad \dot{p}_i = \pdv{L}{q_i},\note{Sostituiamo $p_i=\pdv{L}{\dot{q}_i}$ nelle equazioni di Lagrange e otteniamo $\dot{p}_i = \pdv{L}{q_i}$}
\end{equation}
ottenendo $2n$ equazioni differenziali del primo ordine invece di $n$ equazioni differenziali del secondo ordine.

Calcoliamo il differenziale della lagrangiana e applichiamo la trasformata di Legendre
\begin{align*}
    \dd{L} &= \sum_{j=1}^n \left( \pdv{L}{\dot{q}_j} \dd{\dot{q_j}} + \pdv{L}{q_j} \dd{q_j} \right) + \pdv{L}{t} \dd{t} \\
    &= \sum_{j=1}^n \dot{p}_j \dd{q_j} +  \sum_{j=1}^n p_j \dd{\dot{q_j}} + \pdv{L}{t} \dd{t} \\
    &= \sum_{j=1}^n \dot{p}_j \dd{q_j} +  \dd {\left(\sum_{j=1}^n p_j \dot{q}_j\right)} - \sum_{j=1}^n \dot{q}_j \dd{p_j} + \pdv{L}{t} \dd{t}.
\end{align*}
Portando al primo membro si ha
\begin{equation*}
    \dd \left(L - \sum_{j=1}^n p_j\dot{q}_j \right) = \sum_{j=1}^n \dot{p}_j \dd{q_j} - \sum_{j=1}^n \dot{q}_j \dd{p_j} + \pdv{L}{t} \dd{t}.
\end{equation*}
Definiamo quindi la funzione
\begin{equation} \label{eq:hamiltoniana}
    H(\vq, \vp, t) \defeq \sum_{j=1}^n p_j\dot{q}_j - L(\vq, \dot{\vq}, t),
\end{equation}
ottenendo
\begin{equation*}
    \dd{H} = -\sum_{j=1}^n \dot{p}_j \dd{q_j} + \sum_{j=1}^n \dot{q}_j \dd{p_j} - \pdv{L}{t} \dd{t}.
\end{equation*}
Calcolando il differenziale di $H$ dalla \eqref{eq:hamiltoniana} si ha
\begin{equation*}
    \dd{H} = \sum_{j=1}^n \pdv{H}{q_j} \dd{q_j} + \sum_{j=1}^n \pdv{H}{p_j} \dd{p_j} + \pdv{H}{t} \dd{t},
\end{equation*}
e confrontando le due equazioni otteniamo
\begin{equation*}
 \dot{q}_j = \pdv{H}{p_j}, \qquad \dot{p}_j = -\pdv{H}{q_j}, \qquad \pdv{H}{t} = -\pdv{L}{t}.
\end{equation*}

La funzione $H$ è la trasformata di Legendre di $L$ e viene detta \emph{hamiltoniana} del sistema.

Le $2n$ equazioni differenziali ordinarie del primo ordine in $2n$ incognite
\begin{equation*}
    \begin{cases}
        \displaystyle \dot{q}_j = \pdv{H}{p_j} \\[0.3cm]
        \displaystyle \dot{p}_j = -\pdv{H}{q_j}
    \end{cases}
\end{equation*}
vengono dette \emph{equazioni di Hamilton} o \emph{equazioni canoniche della meccanica}.

Se la matrice jacobiana della trasformazione $\dot{\vq} \mapsto \vp$, nota anche come \emph{matrice hessiana} della lagrangiana rispetto alle velocità, ha determinante non nullo:
\begin{equation*}
    \det\left(\pdv{p_i}{\dot{q}_j}\right) = \det\left(\pdv[2]{L}{\dot{q}_i}{\dot{q}_j}\right) \neq 0,
\end{equation*}
allora, per il teorema della funzione inversa, è possibile invertire le relazioni \eqref{eq:momenticoniugati} ricavando le velocità in funzione dei momenti, $\dot{\vq} = \dot{\vq}(\vq, \vp, t)$. Sostituendo tale espressione si ottiene un'hamiltoniana $H$ che è funzione esclusiva di $\vq$, $\vp$ e $t$.
La trasformazione di Legendre permette quindi il passaggio dalle variabili lagrangiane a quelle hamiltoniane:
\begin{equation*}
    (\vq, \dot{\vq}) \longleftrightarrow (\vq, \vp),
\end{equation*}
ovvero si passa dallo spazio $n$-dimensionale delle configurazioni allo spazio $2n$-dimensionale detto \emph{spazio delle fasi} di Gibbs $\Phi_{2n}$. 

\begin{remark} \label{rem:matrice_hessiana}
    Consideriamo un sistema meccanico a vincoli ideali e scleronomi, in cui le forze attive derivano da un potenziale del tipo ${V=V(\vq, t)}$. Date le ipotesi sui vincoli, dell'energia cinetica sopravvive solo la sua parte quadratica:
    \begin{equation}\label{eq:T_2_quadratica}
        T = T_2 = \frac{1}{2}\sum_{k,h=1}^n a_{kh} \dot{q}_k \dot{q}_h.
    \end{equation}
    Derivando la lagrangiana $L = T - V$ rispetto alle velocità generalizzate $\dot{\vq}$, poiché l'energia potenziale non ne dipende, si ha
    \begin{equation*}
        \pdv[2]{L}{\dot{q}_i}{\dot{q}_j} = \pdv[2]{T_2}{\dot{q}_i}{\dot{q}_j} = \pdv{}{\dot{q}_i} \left( \sum_{k=1}^n a_{jk} \dot{q}_k \right) = \sum_{k=1}^n a_{jk} \delta_{ik} = a_{ji} = a_{ij}.
    \end{equation*}
    Da ciò segue che l'hessiana coincide con la matrice di massa $\tens{A}$. Poiché l'energia cinetica è intrinsecamente una forma quadratica definita positiva, la matrice $\tens{A}$ è sempre invertibile. Questo rende palese l'invertibilità della trasformazione $\dot{\vq} \mapsto \vp$, garantendo che il passaggio allo spazio delle fasi sia sempre lecito e ben definito.
    
    Da tale struttura discende direttamente:
    \begin{equation}\label{eq:p_i_lineari}
        p_i = \pdv{L}{\dot{q}_i} = \pdv{T_2}{\dot{q}_i} = \sum_{j=1}^n a_{ij} \dot{q}_j \implies \vp = \tens{A} \dot{\vq}.
    \end{equation}
\end{remark}

\begin{example}[Oscillatore armonico]
    % Guida orizzontale, molla con coefficiente k nell'origine punto P di massa M
    Consideriamo un oscillatore armonico, ovvero una massa $m$ che si muove su una guida orizzontale e che è collegata a una molla con coefficiente elastico $k$ ancorata all'origine. La coordinata generalizzata è data da $q = x$, e la lagrangiana è data da
    \begin{equation*}
        L = T - V = \frac{1}{2} m \dot{q}^2 - \frac{1}{2} k q^2.
    \end{equation*}

    Il momento coniugato è dato da
    \begin{equation*}
        p = \pdv{L}{\dot{q}} = m \dot{q},
    \end{equation*}
    e l'hamiltoniana è data da
    \begin{equation*}
        H = \left[p \dot{q} - L\right]_{\dot{q} = \frac{1}{m} p} = \frac{p^2}{2m} + \frac{1}{2} k q^2 = \frac{1}{2m}\left(p^2 + mk q^2\right) = \frac{1}{2m}\left(p^2 + m^2\omega^2 q^2\right).
    \end{equation*}
    Le equazioni di Hamilton sono date da
    \begin{equation*}
        \begin{cases}
            \displaystyle \dot{q} = \pdv{H}{p} = \frac{p}{m} \\[0.3cm]
            \displaystyle \dot{p} = -\pdv{H}{q} = -m\omega^2 q
        \end{cases}
    \end{equation*}
\end{example}

\section{L'hamiltoniana e l'energia meccanica}
Sotto le medesime ipotesi dell'osservazione~\ref{rem:matrice_hessiana}, sostituendo l'espressione dei momenti coniugati \eqref{eq:p_i_lineari} nella definizione dell'hamiltoniana si ottiene:
\begin{equation*}
    H = \sum_{j=1}^n p_j \dot{q}_j - L = \sum_{i,j=1}^n a_{ij} \dot{q}_i \dot{q}_j - (T - V).
\end{equation*}
Ricordando la definizione dell'energia cinetica \eqref{eq:T_2_quadratica}, la relazione si riduce a:
\begin{equation*}
    H = 2T - (T - V) = T + V.
\end{equation*}
L'hamiltoniana coincide pertanto con l'energia meccanica totale del sistema.

Un risultato fondamentale che discende dalle equazioni canoniche riguarda l'evoluzione temporale dell'hamiltoniana. Calcoliamo la derivata totale di $H(\vq(t), \vp(t), t)$ rispetto al tempo lungo una traiettoria del sistema:
\begin{equation*}
    \dv{H}{t} = \sum_{j=1}^n \left( \pdv{H}{q_j} \dot{q}_j + \pdv{H}{p_j} \dot{p}_j \right) + \pdv{H}{t}.
\end{equation*}
Sostituendo le equazioni di Hamilton $\dot{q}_j = \pdv{H}{p_j}$ e $\dot{p}_j = -\pdv{H}{q_j}$ all'interno della sommatoria, si ottiene:
\begin{align*}
    \dv{H}{t} &= \sum_{j=1}^n \left( \pdv{H}{q_j} \pdv{H}{p_j} - \pdv{H}{p_j} \pdv{H}{q_j} \right) + \pdv{H}{t} \\
    &= \pdv{H}{t}.
\end{align*}
Questa identità dimostra che la variazione totale dell'hamiltoniana nel tempo coincide con la sua derivata parziale. Di conseguenza, se il sistema è autonomo, ovvero se l'hamiltoniana non dipende esplicitamente dal tempo ($\pdv{H}{t} = 0$), allora $H$ è una costante del moto ($\dv{H}{t} = 0$). Questo risultato rappresenta la formulazione hamiltoniana del teorema di conservazione dell'energia.

\begin{remark}
    Nella formulazione dello spazio delle fasi il tempo può essere trattato come una coordinata generalizzata; in tale contesto, il suo momento coniugato è dato da
    \begin{equation*}
        p_t = - H.
    \end{equation*}
    Per cui, il momento coniugato al tempo coincide con l'energia meccanica totale del sistema, quando quest'ultima coincide con l'hamiltoniana.
\end{remark}

Va notato che la funzione $H$ presenta una forma estremamente simile alla funzione $h$ di Jacobi:
\begin{equation*}
    H(\vq, \vp, t) = \sum_{j=1}^n p_j \dot{q}_j - L(\vq, \dot{\vq}, t), \qquad h(\vq, \dot{\vq}, t) = \sum_{j=1}^n \pdv{L}{\dot{q}_j} \dot{q}_j - L(\vq, \dot{\vq}, t).
\end{equation*}
La differenza però è sostanziale: $H$ è una funzione di $\vq$, $\vp$ e $t$, mentre $h$ è una funzione di $\vq$, $\dot{\vq}$ e $t$.

\section{Variabili cicliche in un sistema hamiltoniano}

\begin{definition}[Coordinata ciclica in un sistema hamiltoniano]\note{Lezione 23/04/2026}
    Una coordinata $q_i$ dicesi \emph{ciclica} per il sistema di hamiltoniana $H(\vq, \vp, t)$ se 
    \begin{equation*}
        \pdv{H}{q_i} = 0.
    \end{equation*}
\end{definition}
Derivando la relazione di definizione di $H$ rispetto a $q_i$ si ha
\begin{equation*}
    \pdv{H}{q_i} = - \pdv{L}{q_i},
\end{equation*}
per cui se $q_i$ è ciclica per $H$ allora è ciclica anche per $L$.

Ipotizziamo ora che in un dato sistema hamiltoniano tutte le coordinate $q_i$ siano cicliche e che $H$ non dipenda esplicitamente dal tempo:
\begin{equation*}
    \pdv{H}{q_i} = 0, \qquad i = 1, \dots, n.
\end{equation*}
In questo caso l'hamiltoniana sarebbe una funzione delle sole $p_i$ e le equazioni di Hamilton sarebbero date da
\begin{equation*}
    \begin{cases}
        \displaystyle \dot{q}_i = \pdv{H}{p_i} \\[0.3cm]
        \displaystyle \dot{p}_i = 0
    \end{cases}
\end{equation*}
per cui i momenti coniugati $p_i$ sarebbero costanti 
\begin{equation*}
    p_i = \text{costante} = \alpha_i, \qquad i = 1, \dots, n.
\end{equation*}
Definiamo la quantità
\begin{equation*}
    \omega_i = \left. \pdv{H}{p_i} \right|_{p_i = \alpha_i},
\end{equation*}
e integrando le equazioni di Hamilton si ha
\begin{align*}
    \int \dot{q}_i \dd{t} &= \int \omega_i \dd{t} \\
    \implies q_i &= \omega_i t + \beta_i, \qquad i = 1, \dots, n.
\end{align*}
Si ha quindi che se è possibile trovare un sistema di coordinate in cui tutte le coordinate siano cicliche allora il sistema è risolvibile per \emph{quadrature}, ovvero è possibile risolvere le equazioni di Hamilton attraverso un numero finito di integrazioni.

\section{Principio di Hamilton modificato}
Partiamo dal principio di Hamilton e sostituiamo l'inversa della trasformazione di Legendre della lagrangiana:
\begin{equation*}
    \var \int_{t_0}^{t_1} L(\vq, \dot{\vq}, t) \dd{t} = \var \int_{t_0}^{t_1} \left( \sum_{j=1}^n p_j \dot{q}_j - H(\vq, \vp, t) \right) \dd{t} = 0.
\end{equation*}
Considerando quindi come lagrangiana la funzione
\begin{equation*}
    L^*(\vq, \vp, \dot{\vq}, \dot{\vp}, t) = \sum_{j=1}^n p_j \dot{q}_j - H(\vq, \vp, t),
\end{equation*}
otteniamo
\begin{align*}
    \var \int_{t_0}^{t_1} L^* \dd{t}
    &= \int_{t_0}^{t_1} \sum_{j=1}^n \left( \pdv{L^*}{q_j} - \dv{t}\pdv{L^*}{\dot{q}_j} \right) \var{q_j} \dd{t}  \\
    &\quad + \int_{t_0}^{t_1} \sum_{j=1}^n \left( \pdv{L^*}{p_j} - \dv{t}\pdv{L^*}{\dot{p}_j} \right) \var{p_j} \dd{t} \\
    &\quad + \left[\sum_{j=1}^n \pdv{L^*}{\dot{q}_j} \var{q_j}+ \sum_{j=1}^n \pdv{L^*}{\dot{p}_j} \var{p_j} \right]_{t_0}^{t_1} = 0.
\end{align*}
Notiamo che le sommatorie negli ultimi due termini si annullano, in quanto $\var{q_j}(t_0) = \var{q_j}(t_1) = 0$ e $\pdv{L^*}{\dot{p}_j} = 0$ per ogni $j$.
Quindi, le equazioni di Eulero--Lagrange per $L^*$ sono
\begin{align*}
    \pdv{L^*}{q_j} - \dv{t}\pdv{L^*}{\dot{q}_j} &= - \pdv{H}{q_j} - \dot{p}_j = 0 \iff \pdv{H}{q_j} = - \dot{p}_j \\
    \pdv{L^*}{p_j} - \dv{t}\pdv{L^*}{\dot{p}_j} &= \dot{q}_j - \pdv{H}{p_j} = 0 \iff \pdv{H}{p_j} = \dot{q}_j
\end{align*}
ottenendo così le equazioni di Hamilton.

\section{Parentesi di Poisson}
\begin{definition}[Parentesi di Poisson]
    Siano $f(\vq, \vp, t)$ e $g(\vq, \vp, t)$ due funzioni di classe $C^1$ nelle variabili canoniche\note{Le variabili canoniche sono le coordinate generalizzate $q_i$ e i momenti coniugati $p_i$}. Si definisce \emph{parentesi di Poisson} di $f$ e $g$
    \begin{equation*}
        \pb{f}{g} \defeq \sum_{j=1}^n \left( \pdv{f}{q_j} \pdv{g}{p_j} - \pdv{f}{p_j} \pdv{g}{q_j} \right).
    \end{equation*}
\end{definition}

Analizziamo ora le proprietà delle parentesi di Poisson. Siano $f(\vq, \vp, t)$, $g(\vq, \vp, t)$ e $h(\vq, \vp, t)$ tre funzioni di classe $C^1$ nelle variabili canoniche, allora
\begin{enumerate}[label=($\roman*$)]
    \item $\pb{f}{f} = 0$;
    \item $\pb{f}{g} = - \pb{g}{f}$ (antisimmetria);
    \item $\pb{c}{f} = 0$, per ogni $c\in\R$;
    \item $\pb{\alpha f + \beta g}{h} = \alpha \pb{f}{h} + \beta \pb{g}{h}$, per ogni $\alpha, \beta\in \R$  (linearità);
    \item $\displaystyle \pdv{t} \pb{f}{g} = \pb{\pdv{f}{t}}{g} + \pb{f}{\pdv{g}{t}}$;
    \item $\pb{f}{\pb{g}{h}} + \pb{g}{\pb{h}{f}} + \pb{h}{\pb{f}{g}} = 0$ (identità di Jacobi--Poisson).\note{permutando ciclicamente le funzioni}
\end{enumerate}

\vspace{1em}
È interessante notare che la proprietà ($v$) è valida considerando qualsiasi operatore differenziale $\mathcal{D}$ al posto della derivata temporale $\pdv{t}$:
\begin{equation*}
    \mathcal{D} \pb{f}{g} = \pb{\mathcal{D} f}{g} + \pb{f}{\mathcal{D} g}.
\end{equation*}

\begin{definition}[Integrale primo per le equazioni di Hamilton]
    Una funzione $f(\vq, \vp, t)$ è un \emph{integrale primo} per le equazioni di Hamilton se $f$ si conserva costante sulle soluzioni di tali equazioni.
\end{definition}
\begin{proposition}
    $f$ è un integrale primo se e soltanto se 
    \begin{equation}\label{eq:integrale_primo_poisson}
        \pdv{f}{t} + \pb{f}{H} = 0.
    \end{equation}
\end{proposition}
\begin{proof}
    Data una soluzione $\vq(t)$, $\vp(t)$ delle equazioni di Hamilton, si ha che
    \begin{align*}
        \dv{f(\vq(t), \vp(t), t)}{t} &= \pdv{f}{t} + \sum_{j=1}^n \left( \pdv{f}{q_j} \dot{q}_j + \pdv{f}{p_j} \dot{p}_j \right) \\
        &= \pdv{f}{t} + \sum_{j=1}^n \left(\pdv{f}{q_j} \pdv{H}{p_j} - \pdv{f}{p_j} \pdv{H}{q_j} \right) \\
        &= \pdv{f}{t} + \pb{f}{H}.
    \end{align*}
\end{proof}

\begin{theorem}[Jacobi--Poisson]
    Se $f$ e $g$ sono integrali primi delle equazioni di Hamilton, allora anche $\pb{f}{g}$ è un integrale primo.
\end{theorem}
\begin{proof}
    Vogliamo dimostrare che 
    \begin{equation*}
        \pb{f}{g} \text{ è un integrale primo} \iff \pdv{t} \pb{f}{g} + \pb{\pb{f}{g}}{H} = 0.
    \end{equation*}
    Calcoliamo
    \begin{align*}
        \pdv{t} \pb{f}{g} + \pb{\pb{f}{g}}{H} &= \pb{\pdv{f}{t}}{g} + \pb{f}{\pdv{g}{t}} + \pb{\pb{f}{g}}{H} \\
        &= -\pb{\pb{f}{H}}{g} - \pb{f}{\pb{g}{H}} - \pb{H}{\pb{f}{g}} \note{Utilizzando l'equazione~\eqref{eq:integrale_primo_poisson}}\\
        &= -\pb{g}{\pb{H}{f}} - \pb{f}{\pb{g}{H}} - \pb{H}{\pb{f}{g}} \\
        &= 0,
    \end{align*}
    avendo utilizzato l'identità di Jacobi--Poisson.
\end{proof}

\begin{definition}
    Si dicono \emph{parentesi di Poisson fondamentali} le seguenti relazioni fra le variabili canoniche $q_i$ e $p_i$:
    \begin{itemize}
        \item $\pb{q_i}{q_j} = 0$;
        \item $\pb{p_i}{p_j} = 0$;
        \item $\pb{q_i}{p_j} = \delta_{ij}$,
    \end{itemize}
    dove $\delta_{ij}$ è la delta di Kronecker.
\end{definition}

Calcolando le parentesi di Poisson fra $q_i$ e $H$ e fra $p_i$ e $H$ si ha
\begin{equation*}
    \pb{q_i}{H} = \pdv{H}{p_i}, \qquad \pb{p_i}{H} = - \pdv{H}{q_i},
\end{equation*}
per cui le equazioni di Hamilton possono essere riscritte in termini di parentesi di Poisson
\begin{equation*}
    \begin{cases}
        \displaystyle \dot{q}_i = \pb{q_i}{H} \\
        \displaystyle \dot{p}_i = \pb{p_i}{H}
    \end{cases}
\end{equation*}

\section{Elementi dell'approccio geometrico simplettico}
Introduciamo la quantità\note{Lezione 28/04/2026}
\begin{equation*}
    \eta_i = q_i, \qquad \eta_{n+i} = p_i, \qquad i = 1, \dots, n,
\end{equation*}
definendo un vettore $\vb*{\eta} = (q_1, \dots, q_n, p_1, \dots, p_n)$ che contiene tutte le variabili canoniche. Definiamo inoltre la matrice
\begin{align*}
    \tens{J} &= \begin{pmatrix}
        \tens{0} & \tens{1} \\
        -\tens{1} & \tens{0}
    \end{pmatrix} \\
    &= \begin{pmatrix}
        0 & \dots & 0 & 1 & \dots & 0 \\
        \vdots & \ddots & \vdots & \vdots & \ddots & \vdots \\
        0 & \dots & 0 & 0 & \dots & 1 \\
        -1 & \dots & 0 & 0 & \dots & 0 \\
        \vdots & \ddots & \vdots & \vdots & \ddots & \vdots \\
        0 & \dots & -1 & 0 & \dots & 0
    \end{pmatrix}
\end{align*}
detta \emph{matrice simplettica}, dove $\tens{0}$ è la matrice nulla $n\times n$ e $\tens{1}$ è la matrice identità $n\times n$. Si ha che per questa matrice $\tens{J}^T = \tens{J}^{-1} = -\tens{J}$ e $\det \tens{J} = 1$.

Le equazioni di Hamilton possono essere riscritte in forma compatta
\begin{equation*}
    \dot{\vb{\eta}} = \tens{J} \grad_{\vb{\eta}} H.
\end{equation*}

\section{Trasformazioni canoniche}
Le equazioni di Lagrange sono invarianti per trasformazioni di coordinate. L'elevata simmetria delle equazioni di Hamilton rende più complicato questo problema: può accadere che partendo da un sistema hamiltoniano e applicando una trasformazione di coordinate, il sistema trasformato non è più hamiltoniano. 

Come per l'approccio lagrangiano, è fondamentale che la trasformazione di coordinate sia regolare, ovvero che 
\begin{equation*}
\det \pdv{(Q_1, \dots, Q_n, P_1, \dots, P_n)}{(q_1, \dots, q_n, p_1, \dots, p_n)} \neq 0,
\end{equation*}
tuttavia ciò non è sufficiente per garantire che le nuove coordinate trasformate rispettino le equazioni di Hamilton.

\begin{definition}[Trasformazione canonica]
    Una trasformazione di coordinate $Q_i = Q_i(\vq, \vp, t)$, $P_i=P_i(\vq,\vp, t)$ nello spazio delle fasi si dice \emph{canonica} se conserva la struttura canonica delle equazioni, ovvero se le nuove coordinate $Q_i$ e $P_i$ soddisfano le equazioni di Hamilton con la nuova hamiltoniana $K(\vQ, \vP, t)$:
    \begin{equation*}
        \begin{cases}
            \displaystyle \dot{Q}_i = \pdv{K}{P_i} \\[0.3cm]
            \displaystyle \dot{P}_i = -\pdv{K}{Q_i}
        \end{cases}
    \end{equation*}

    Inoltre, una trasformazione canonica di coordinate indipendente dal tempo si dirà \emph{completamente canonica}.
\end{definition}

Data una trasformazione canonica, il principio di Hamilton modificato per $H$ e $K$ si scrive come
\begin{align*}
    \var \int_{t_0}^{t_1} \left( \sum_{j=1}^n p_j \dot{q}_j - H(\vq, \vp, t) \right) \dd{t} &= 0 \\
    \iff \var \int_{t_0}^{t_1} \left( \sum_{j=1}^n P_j \dot{Q}_j - K(\vQ, \vP, t) \right) \dd{t} &= 0.
\end{align*}

\subsection{Condizioni di canonicità}
Una condizione necessaria e sufficiente affinché una trasformazione di coordinate sia canonica è data dal seguente teorema.

\begin{theorem}[Condizione di canonicità di Lie]
    Una trasformazione puntuale nello spazio delle fasi,
    \begin{equation*}
        Q_i = Q_i(\vq, \vp, t), \qquad P_i = P_i(\vq, \vp, t), \qquad i = 1, \dots, n,
    \end{equation*}
    è canonica se e solo se 
    \begin{equation*}
        \lambda \left(\sum_{j=1}^n p_j \dot{q}_j - H(\vq, \vp, t)\right) = \sum_{j=1}^n P_j \dot{Q}_j - K(\vQ, \vP, t) + \dv{F}{t},
    \end{equation*}
    dove $\lambda$ è una costante reale non nulla, detta \emph{valenza della trasformazione}, e $F$ è una funzione delle $2n$ variabili canoniche e del tempo, detta \emph{funzione generatrice della trasformazione canonica}.
\end{theorem}

L'aggiunta del termine $\dv{F}{t}$ all'integrando non ha effetto sulle equazioni di Eulero--Lagrange, poiché la variazione prima del suo integrale è identicamente nulla. Infatti, esplicitando l'operatore di variazione $\var$ si ottiene
\begin{align*}
    \var \int_{t_0}^{t_1} \dv{F}{t} \dd{t} &= \var \left[ F(\vq, \vp, t) \right]_{t_0}^{t_1} \\
    &= \left[ \sum_{j=1}^n \left(\pdv{F}{q_j} \var{q_j} + \pdv{F}{p_j} \var{p_j} \right) \right]_{t_0}^{t_1} \\
    &= \sum_{j=1}^n \left( \left. \pdv{F}{q_j} \right|_{t_1} \var{q_j}(t_1) - \left. \pdv{F}{q_j} \right|_{t_0} \var{q_j}(t_0) \right) \\
    &\quad + \sum_{j=1}^n \left( \left. \pdv{F}{p_j} \right|_{t_1} \var{p_j}(t_1) - \left. \pdv{F}{p_j} \right|_{t_0} \var{p_j}(t_0) \right).
\end{align*}
Ricordando che $\var{\vq}(t_0) = \var{\vq}(t_1) = \vb{0}$ e $\var{\vp}(t_0) = \var{\vp}(t_1) = \vb{0}$, si conclude che
\begin{equation*}
    \var \int_{t_0}^{t_1} \dv{F}{t} \dd{t} = 0.
\end{equation*}

D'ora in avanti considereremo solo le trasformazioni canoniche con valenza $\lambda = 1$, dette \emph{univalenti}.

\begin{remark}
    Una trasformazione canonica non è legata a una specifica hamiltoniana, ma è canonica per qualsiasi sistema.
\end{remark}

La condizione di canonicità si può anche esprimere in forma differenziale:
\begin{equation*}
    \sum_{j=1}^n p_j \dd{q_j} - H(\vq, \vp, t) \dd{t} = \sum_{j=1}^n P_j \dd{Q_j} - K(\vQ, \vP, t) \dd{t} + \dd{F}.
\end{equation*}

Ulteriori condizioni per la canonicità di una trasformazione di coordinate sono le seguenti.

\begin{proposition}
    Una trasformazione di coordinate $Q_i = Q_i(\vq, \vp, t)$, $P_i = P_i(\vq, \vp, t)$ è canonica se e solo se conserva le parentesi di Poisson fondamentali, ovvero se
    \begin{equation*}
        \pb{Q_i}{Q_j} = 0, \qquad \pb{P_i}{P_j} = 0, \qquad \pb{Q_i}{P_j} = \delta_{ij}.
    \end{equation*}
\end{proposition}

\begin{proposition}
    Una trasformazione di coordinate $Q_i = Q_i(\vq, \vp, t)$, $P_i = P_i(\vq, \vp, t)$ è canonica se e solo se
    \begin{equation*}
        \det \pdv{(Q_1, \dots, Q_n, P_1, \dots, P_n)}{(q_1, \dots, q_n, p_1, \dots, p_n)} = 1.
    \end{equation*}
\end{proposition}
\begin{remark}
    La condizione di canonicità impone una precisa restrizione algebrica sulla matrice Jacobiana della trasformazione, che indicheremo con $\tens{M}$. Essa deve infatti preservare la struttura della matrice simplettica $\tens{J}$, traducendosi nella relazione matriciale
    \begin{equation*}        
        \tens{M}^T \tens{J} \tens{M} = \tens{J}.
    \end{equation*}
    Si noti come, applicando il teorema di Binet a questa espressione e ricordando che $\det\tens{J} = 1$, si ricavi immediatamente la condizione necessaria $\det\tens{M} = 1$.
\end{remark}

\subsection{Costruzione di trasformazioni canoniche fondamentali mediante funzioni generatrici}

Presentiamo ora quattro trasformazioni canoniche notevoli, ottenute mediante successive trasformate di Legendre della funzione generatrice. Queste costruzioni mostrano come il principio di Lie fornisce un metodo sistematico per generare trasformazioni canoniche di varia tipologia.

\paragraph{Generatrice $F_1(\vq, \vQ, t)$}
Consideriamo una trasformazione canonica con funzione generatrice $F=F_1(\vq, \vQ, t)$. Scrivendo la forma differenziale della condizione di Lie si ottiene
\begin{multline*}
    \sum_{j=1}^n p_j \dd{q_j} - H(\vq, \vp, t) \dd{t} = \sum_{j=1}^n P_j \dd{Q_j} - K(\vQ, \vP, t) \dd{t} \\
    + \sum_{j=1}^n \pdv{F_1}{q_j} \dd{q_j} + \sum_{j=1}^n \pdv{F_1}{Q_j} \dd{Q_j} + \pdv{F_1}{t} \dd{t}.
\end{multline*}
Portando tutto a primo membro e raccogliendo rispetto a $\dd{q_j}$, $\dd{Q_j}$ e $\dd{t}$, l'equazione assume la forma
\begin{equation*}
    \sum_{j=1}^n \left(p_j - \pdv{F_1}{q_j}\right) \dd{q_j} - \sum_{j=1}^n \left(P_j + \pdv{F_1}{Q_j}\right) \dd{Q_j} - \left(H - K + \pdv{F_1}{t}\right) \dd{t} = 0.
\end{equation*}
Affinché questa identità sia soddisfatta, i coefficienti dei differenziali devono annullarsi identicamente. Si ottengono quindi le relazioni
\begin{equation*}
    p_j = \pdv{F_1}{q_j}, \qquad
    P_j = - \pdv{F_1}{Q_j}, \qquad
    K = H + \pdv{F_1}{t}.
\end{equation*}

\paragraph{Generatrice $F_2(\vq, \vP, t)$}
Costruiamo questa generatrice applicando la trasformata di Legendre a $F=F_1$ rispetto alla variabile $Q_j$. Partendo dalla condizione di Lie e applicando la regola di Leibniz, si ha
\begin{align*}
    \sum_{j=1}^n p_j \dd{q_j} - H \dd{t} &= \sum_{j=1}^n P_j\dd{Q_j} - K\dd{t} + \dd{F} \\
    &= \dd{\left(\sum_{j=1}^n P_j Q_j\right)} - \sum_{j=1}^n Q_j \dd{P_j} - K\dd{t} + \dd{F}.
\end{align*}
Raggruppando i termini si ha
\begin{equation*}
    \sum_{j=1}^n p_j \dd{q_j} + \sum_{j=1}^n Q_j \dd{P_j} 
    + (K - H)\dd{t} = \dd{\left(F + \sum_{j=1}^n P_j Q_j\right)}.
\end{equation*}
La funzione generatrice $F_2$ è quindi data da
\begin{equation*}
    F_2(\vq, \vP, t) = \left[ F(\vq, \vQ, t) + \sum_{j=1}^n P_j Q_j \right]_{\vQ = \vQ(\vP, \vq, t)}.
\end{equation*}
Dalle corrispondenze tra i differenziali indipendenti $\dd{q_j}$, $\dd{P_j}$, $\dd{t}$, si ottiene
\begin{equation*}
    p_j = \pdv{F_2}{q_j}, \qquad
    Q_j = \pdv{F_2}{P_j}, \qquad
    K = H + \pdv{F_2}{t}.
\end{equation*}

\paragraph{Generatrice $F_3(\vp, \vQ, t)$}
Questa generatrice si ottiene mediante trasformata di Legendre di $F=F_1$ rispetto a $q_j$. Partendo dalla condizione di Lie e applicando la regola di Leibniz per il differenziale $\dd(\sum p_j q_j)$, si ha
\begin{equation*}
    \dd{\left(\sum_{j=1}^n p_j q_j\right)} - \sum_{j=1}^n q_j \dd{p_j} - H\dd{t} = \sum_{j=1}^n P_j \dd{Q_j} - K \dd{t} + \dd{F}.
\end{equation*}
Raggruppando i termini si ha
\begin{equation*}
    - \sum_{j=1}^n q_j \dd{p_j} - \sum_{j=1}^n P_j \dd{Q_j} + (K - H)\dd{t} = \dd{\left(F - \sum_{j=1}^n p_j q_j\right)}.
\end{equation*}
La funzione generatrice $F_3$ è quindi data da
\begin{equation*}
    F_3(\vp, \vQ, t) = \left[ F(\vq, \vQ, t) - \sum_{j=1}^n p_j q_j \right]_{\vq = \vq(\vp, \vQ, t)}.
\end{equation*}
Dalle corrispondenze tra i differenziali indipendenti $\dd{p_j}$, $\dd{Q_j}$, $\dd{t}$, si ottiene
\begin{equation*}
    q_j = - \pdv{F_3}{p_j}, \qquad
    P_j = -\pdv{F_3}{Q_j}, \qquad
    K = H + \pdv{F_3}{t}.
\end{equation*}

\paragraph{Generatrice $F_4(\vp, \vP, t)$}
La generatrice $F_4$ si ottiene applicando trasformate di Legendre sia rispetto a $q_j$ che rispetto a $Q_j$. Partendo dalla condizione di Lie e applicando la regola di Leibniz per i differenziali $\dd(\sum p_j q_j)$ e $\dd(\sum P_j Q_j)$, si ha
\begin{equation*}
    \dd{\left(\sum_{j=1}^n p_j q_j\right)} - \sum_{j=1}^n q_j \dd{p_j} - H\dd{t} = \dd{\left(\sum_{j=1}^n P_j Q_j\right)} - \sum_{j=1}^n Q_j \dd{P_j} - K \dd{t} + \dd{F}.
\end{equation*}
Raggruppando i termini si ha
\begin{equation*}
    - \sum_{j=1}^n q_j \dd{p_j} + \sum_{j=1}^n Q_j \dd{P_j} + (K - H)\dd{t} = \dd\left(F - \sum_{j=1}^n p_j q_j + \sum_{j=1}^n P_j Q_j\right).
\end{equation*}
La funzione generatrice $F_4$ è quindi data da
\begin{equation*}
    F_4(\vp, \vP, t) = \left[ F(\vq, \vQ, t) - \sum_{j=1}^n p_j q_j + \sum_{j=1}^n P_j Q_j \right]_{\substack{\vQ = \vQ(\vp, \vP, t) \\ \vq = \vq(\vp, \vP, t)}}.
\end{equation*}
Dalle corrispondenze tra i differenziali indipendenti $\dd{p_j}$, $\dd{P_j}$, $\dd{t}$, si ottiene
\begin{equation*}
    q_j = -\pdv{F_4}{p_j}, \qquad
    Q_j = \pdv{F_4}{P_j}, \qquad
    K = H + \pdv{F_4}{t}.
\end{equation*}

Questi quattro tipi di funzioni generatrici rappresentano i casi più comuni e naturali. Il concetto di trasformazione canonica è tuttavia molto più generale: il vincolo essenziale è che la trasformazione soddisfi la \emph{condizione di Lie}. In linea di principio, nulla ci impedisce di costruire generatrici che dipendono da tutte e quattro le variabili canoniche $p_i$, $q_i$, $P_i$, $Q_i$ e dal tempo $t$ o da qualsiasi combinazione di esse. La scelta della generatrice più opportuna è dettata dalla convenienza computazionale e dalla natura del problema da risolvere.

\begin{example}[Oscillatore armonico] \label{ex:oscillatore-armonico-hamilton}
    L'hamiltoniana dell'oscillatore armonico è
    \begin{equation*}
        H(q, p) = \frac{1}{2m}(p^2 + m^2 \omega^2 q^2).
    \end{equation*}
    Consideriamo la seguente trasformazione
    \begin{equation*}
        q = \frac{f(P)}{m\omega} \sin Q, \qquad p = f(P) \cos Q .
    \end{equation*}
    Se questa trasformazione fosse canonica avremmo un'hamiltoniana del tipo
    \begin{equation*}
        H = \frac{1}{2m}\left[f^2(P) \cos^2 Q + m^2 \omega^2 f^2(P) \sin^2 Q\right] = \frac{f^2(P)}{2m},
    \end{equation*}
    che è indipendente da $Q$, rendendo il sistema integrabile.

    Analizziamo ora se questa trasformazione è effettivamente canonica per qualche scelta di $f$. Consideriamo la funzione generatrice
    \begin{equation*}
        F_1(q, Q) = \frac{m\omega q^2}{2}\cot Q.
    \end{equation*}
    Per cui
    \begin{equation*}
        p = \pdv{F_1}{q} = m\omega q \cot Q, \qquad P = -\pdv{F_1}{Q} = \frac{m\omega q^2}{2\sin^2 Q}.
    \end{equation*}
    Da cui si ricava
    \begin{equation*}
        q = \sqrt{\frac{2P}{m\omega}} \sin Q, \qquad p = \sqrt{2m\omega P} \cos Q,
    \end{equation*}
    ottenendo così la trasformazione desiderata con $f(P) = \sqrt{2m\omega P}$, e quindi
    \begin{equation*}
        H = \frac{f^2(P)}{2m} = \omega P.
    \end{equation*}

    Le equazioni di Hamilton per questa hamiltoniana sono
    \begin{equation*}
        \begin{cases}
            \displaystyle \dot{Q} = \pdv{H}{P} = \omega \\[0.3cm]
            \displaystyle \dot{P} = -\pdv{H}{Q} = 0
        \end{cases}
    \end{equation*}
    da cui si ricava la soluzione generale
    \begin{equation*}
        Q(t) = \omega t + \alpha, \qquad P(t) = \text{costante} = \frac{E}{\omega}.
    \end{equation*}
    Per cui, ritornando alle variabili originali, si ha
    \begin{equation*}
        q(t) = \sqrt{\frac{2E}{m\omega^2}} \sin(\omega t + \alpha), \qquad p(t) = \sqrt{2m\omega E} \cos(\omega t + \alpha).
    \end{equation*}
    Si nota che questa forma è molto più complessa della soluzione nelle coordinate trasformate.
\end{example}
